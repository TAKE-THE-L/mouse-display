const extension = class {

constructor() {

this.name = "Near Mouse Display"; // Extension Name

}

// Block to display text near the mouse

getBlocks() {

return [

{

opcode: 'displayTextNearMouse',

blockType: TurboWarp.BlockType.REPORTER,

text: 'display text near mouse',

func: this.displayTextNearMouse,

},

];

}

// Function that runs when the block is executed

displayTextNearMouse() {

// Get the mouse's current position in TurboWarp

const mouseX = TurboWarp.vm.mouseX;

const mouseY = TurboWarp.vm.mouseY;

const distance = 10; // Distance from the mouse pointer where we want to place the text

const newX = mouseX + distance; // Adjust X position

const newY = mouseY + distance; // Adjust Y position

// Create a new sprite for displaying the text

const sprite = TurboWarp.vm.createSprite('textSprite');

// Set the costume to plain text (you may want to add a plain text costume or use an existing one)

const costume = new TurboWarp.Costume('plainText'); // Make sure the costume exists

sprite.setCostume(costume);

sprite.setSize(50); // Adjust size of the text

sprite.setXY(newX, newY); // Position the sprite near the mouse

// Return the text (or string) to be displayed

return 'Hello!';

}

};

// Register the TurboWarp extension

TurboWarp.registerExtension(extension);
